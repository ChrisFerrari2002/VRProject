216void ENG_API Eng::Base::reshapeCallback(int width, int height)
217{
218 glViewport(0, 0, width, height);
219 glMatrixMode(GL_PROJECTION);
220 //create a perspective matrix with a 45 degree field of view and a near and far plane
221 perspective = glm::perspective(glm::radians(80.0f), (float)width / (float)height, 1.0f, 1000.0f);
222 ortho = glm::ortho(0.0f, (float)width, 0.0f, (float)height, 1.0f, -1.0f);

397void Eng::Base::setObjectPickedCallback(void (*func)(Node* n, bool mousePressed)) {
398 static std::function<void(int, int)> lambdaWrapper = [func](int x, int y) {
399 if (x == -1 && y == -1)
400 return func(nullptr, false);
401
402 glDisable(GL_LIGHTING);
403 glDisable(GL_TEXTURE_2D);
404 glViewport(0, 0, glutGet(GLUT_WINDOW_WIDTH), glutGet(GLUT_WINDOW_HEIGHT));
405
406 glMatrixMode(GL_PROJECTION);
407 glLoadMatrixf(glm::value_ptr(perspective));
408 glMatrixMode(GL_MODELVIEW);
409 glLoadMatrixf(glm::value_ptr(cameras.at(activeCamera)->getInverseCameraMat()));
410
411 list.render(cameras.at(activeCamera)->getInverseCameraMat(), (void*)true);
412
413
414 unsigned char pixel[4];
415 glReadPixels(x, glutGet(GLUT_WINDOW_HEIGHT) - y, 1, 1, GL_RGBA, GL_UNSIGNED_BYTE, pixel);
416
417 glEnable(GL_LIGHTING);
418 glEnable(GL_TEXTURE_2D);
419
420 int id = (pixel[0] << 16) | (pixel[1] << 8) | (pixel[2] << 0);
421 Node* n = list.getObjectById(id);
422 if (n != nullptr) {
423 return func(n, true);
424 }
425 return func(nullptr, false);
426 };
427
428 //call lambdaWrapper when mouse left button is pressed
429 glutMouseFunc([](int button, int state, int x, int y) {
430 if (button == GLUT_LEFT_BUTTON && state == GLUT_DOWN) {
431 Eng::Base::clearWindow();
432 lambdaWrapper(x, y);
433 }
434 if (button == GLUT_LEFT_BUTTON && state == GLUT_UP) {
435 lambdaWrapper(-1, -1);
436 }
437 });
438}
