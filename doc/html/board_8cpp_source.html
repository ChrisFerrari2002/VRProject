48std::pair<int, int> Board::getPosition(const std::string& nodeName) {
49 // Regex per identificare i pezzi (p_<color>_<type><number>)
50 std::regex pieceRegex("p_(white|black)_(rook|horse|bishop|queen|king|pawn)([0-9]*)");
51 std::smatch match;
52
53 if (std::regex_match(nodeName, match, pieceRegex)) {
54 std::string color = match[1];
55 std::string type = match[2];
56 int number = match[3].length() > 0 ? std::stoi(match[3]) : 0;
57
58 if (type == "rook") {
59 return (color == "white") ? std::make_pair(0, number == 1 ? 0 : 7) : std::make_pair(7, number == 1 ? 7 : 0);
60 }
61 else if (type == "horse") {
62 return (color == "white") ? std::make_pair(0, number == 1 ? 1 : 6) : std::make_pair(7, number == 1 ? 1 : 6);
63 }
64 else if (type == "bishop") {
65 return (color == "white") ? std::make_pair(0, number == 1 ? 2 : 5) : std::make_pair(7, number == 1 ? 5 : 2);
66 }
67 else if (type == "queen") {
68 return (color == "white") ? std::make_pair(0, 4) : std::make_pair(7, 4);
69 }
70 else if (type == "king") {
71 return (color == "white") ? std::make_pair(0, 3) : std::make_pair(7, 3);
72 }
73 else if (type == "pawn") {
74 return (color == "white") ? std::make_pair(1, number - 1) : std::make_pair(6, number - 1);
75 }
76 }
77
78 return std::make_pair(-1, -1); // Posizione invalida
79}

172void Board::confirmMovement() {
173 if (selectedPiece == nullptr) return;
174 if (selectedPiece->isEliminated()) return;
175
176 int pieceRow = selectedPiece->getTempRow();
177 int pieceCol = selectedPiece->getTempCol();
178 int initialPieceRow = selectedPiece->getRow();
179 int initialPieceCol = selectedPiece->getCol();
180
181 // Destination empty
182 if (table[pieceRow][pieceCol] == nullptr) {
183 table[initialPieceRow][initialPieceCol] = nullptr;
184 table[pieceRow][pieceCol] = selectedPiece;
185 Move* move = new Move(selectedPiece);
186 addMove(move);
187 selectedPiece->setNewPosition(pieceRow, pieceCol);
188 return;
189 }
190
191 Move* move = new Move(selectedPiece);
192 addMove(move);
193
194 // Destination same color
195 if (table[pieceRow][pieceCol]->getColor() == selectedPiece->getColor()) {
196 selectedPiece->resetPosition();
197 return;
198 }
199
200 Piece* capturedPiece = table[selectedPiece->getTempRow()][selectedPiece->getTempCol()];
201 Move* capturedMove = new Move(capturedPiece);
202 capturePiece(capturedPiece);
203 capturedMove->setEliminated(true);
204 capturedMove->setEndPosition(-1, -1);
205 addMove(capturedMove);
206
207 // Move piece to new position
208 table[initialPieceRow][initialPieceCol] = nullptr;
209 table[pieceRow][pieceCol] = selectedPiece;
210 selectedPiece->setNewPosition(pieceRow, pieceCol);
211}

335void Board::redo() {

337
