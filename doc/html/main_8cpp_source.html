83void makeObjectBlink(Eng::Node* obj) {
84 if (!(obj->getName().substr(0, 1) == "p")) return;
85
86 step = 0.01f;
87
88 if (blink) {
89 blinkStep += step;
90 if (blinkStep > range)
91 blink = false;
92 }
93 else {
94 blinkStep -= step;
95 if (blinkStep < 0.0f)
96 blink = true;
97 }
98
99 ((Eng::Mesh*)obj)->getMaterial()->setEmission(glm::vec4(blinkStep * 5.0f, 0.0f, 0.0f, 1.0f));
100}

117void rotateCamera(int value) {

142void getPickedObject(Eng::Node* n, bool mousePressed) {
143 moveConfirmed = false;
144 if (n == nullptr || n->getName().substr(0, 1) != "p") return;
145 if (mousePressed) {
146 if (pickedObject != nullptr) {
147 if (!moveConfirmed) {
148 chessboard.getSelectedPiece()->resetPosition();
149 }
150 ((Eng::Mesh*)pickedObject)->getMaterial()->setEmission(lastObjectEmission);
151 }
152 lastObjectEmission = ((Eng::Mesh*)n)->getMaterial()->getEmission();
153 pickedObject = n;
154 chessboard.setSelectedPiece(pickedObject);
155
156 // Start blinker timer if it isn't already running
157 if (!blinkerTimerStarted) {
158 eng.startTimer(updateBlinking, 10);
159 blinkerTimerStarted = true;
160 }
